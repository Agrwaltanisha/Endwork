
#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

int main(int argc, char** argv) {
    int m, rank, size, i;
    int *data, *local_data;
    double local_sum = 0, total_sum = 0, avg = 0;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    if (rank == 0) {
        printf("Enter the value of m: ");
        scanf("%d", &m);

        data = (int*)malloc(size * m * sizeof(int));

        for (i = 0; i < size * m; i++) {
            printf("Enter element %d: ", i+1);
            scanf("%d", &data[i]);
        }
    }

    MPI_Bcast(&m, 1, MPI_INT, 0, MPI_COMM_WORLD);

    local_data = (int*)malloc(m * sizeof(int));

    MPI_Scatter(data, m, MPI_INT, local_data, m, MPI_INT, 0, MPI_COMM_WORLD);

    for (i = 0; i < m; i++) {
        local_sum += local_data[i];
    }

    avg = local_sum / m;

    MPI_Reduce(&avg, &total_sum, 1, MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);

    if (rank == 0) {
        double total_avg = total_sum / (size * 1.0);
        printf("The total average is: %lf\n", total_avg);
    }

    MPI_Finalize();

    return 0;
}
